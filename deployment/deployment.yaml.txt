apiVersion: apps/v1

metadata:
  name: ngutiwebapp-deployment
spec:
  replicas: 2
  strategy:
    type: Recreate
  selector:
    matchLabels:
      app: ngutiwebapp
  template:
    metadata:
      labels:
        app: ngutiwebapp
    spec:
      containers:
        - name: ngutiwebapp-container
          image: ekangaki/ngutiwebapp:latest
          imagePullPolicy: Always
          ports:
            - containerPort: 8080

kind: Service
metadata:
  name: ngutiwebapp-svc
spec:
  type: LoadBalancer
  selector:
    app: ngutiwebapp
  ports:
    - port: 80
      targetPort: 8080
